    实验1 循环结构：求阶乘末尾零个数

    算法思路：

    • 末尾0的个数取决于阶乘中5的因子的个数，故Z（N）等于阶乘中5的个数，只需写个For循环即可

    • 复杂度 O（n）

    #include <stdio.h>

int Function(int N) {
    int count = 0;
    // 由于末尾零的个数取决于阶乘中5的因子个数，因此只需计算阶乘中5的个数即可
      for (int i = 5; N / i >= 1; i *= 5) {
        count += N / i;
    }
    return count;
}

int main() {
    int T;
    printf("请输入数字的个数：");
    scanf("%d", &T);

    int results[T];
    for (int i = 0; i < T; ++i) {
        int N;
        printf("请输入一个正整数N：");
        scanf("%d", &N);
        results[i] = Function(N);
    }

    // 输出结果
    printf("计算结果：\n");
    for (int i = 0; i < T; ++i) {
        printf("%d\n", results[i]);
    }

    return 0;
}

    实验2循环结构：求阶乘

    算法思路(大数相乘)

    • 注意到N在1000000000以内，直接for循环肯定会爆掉
    • 第一步，创建一个数组来存储大数，数组的大小足够大以容纳阶乘的结果
    • 第二步，实现大数相乘的函数乘法，模拟手算的流程
    • 第三步，调用第二步实现的大数相乘算法，由2迭代到n，每次都将当前数字与之前的结果相乘
    #include <stdio.h>
#include <stdlib.h>
#define MAX_LONG 100000

void multiply(int res[], int x, int *res_size) {
    int carry = 0; //初始化进位为0
	//逐项相乘，模拟手算乘法的过程，并处理进位
    for (int i = 0; i < *res_size; i++) {
        int prod = res[i] * x + carry; //当前位相乘并加上进位
        res[i] = prod % 10; //存储结果的个位数
        carry = prod / 10;// 计算新的进位
    }
// 处理最高位的进位
    while (carry) {
        res[*res_size] = carry % 10;// 存储结果的下一位数
        carry /= 10; // 计算新的进位
        (*res_size)++; // 更新结果的位数
    }
}

void Ans(int N) {
    int res[MAX_LONG] = {0}; //初始化大数数组
    res[0] = 1; //初试阶乘结果为1
    int res_size = 1;//初始化数组长度1

    for (int x = 2; x <= N; x++)
        multiply(res, x, &res_size);

    // 输出结果
    printf("%.4e\n");
}

int main() {
    int T;
    printf("请输入数字的个数：");
    scanf("%d", &T);

    for (int i = 0; i < T; ++i) {
        int N;
        printf("请输入一个正整数N：");
        scanf("%d", &N);
        Ans(N);
    }

    return 0;
}

    优化思路

    • #include <stdio.h>
#include <math.h>

double Ans(int N) {
    double ans = 0; // 初始化结果为0
    for (int i = 1; i <= N; ++i) {
        ans += log10(i); // 累加每个数字的以10为底的对数
    }
    return ans;
}

int main() {
    int T;
    printf("请输入数字的个数：");
    scanf("%d", &T);

    for (int i = 0; i < T; ++i) {
        int N;
        printf("请输入一个正整数N：");
        scanf("%d", &N);
        double temp = Ans(N);
        double intPart, fracPart;
        fracPart = modf(temp, &intPart); // 将小数与整数分离
        double result = pow(10, fracPart);
        printf("%.4lf e+%d\n", result, (int)intPart); // 输出结果
    }

    return 0;
}

    实验三 组合数分解

    算法思路：

    • **分解算法：**首先对于给定N和k，我们需要找到最大的j满足C（j,i）<=N，找到这样的j之后，计算其组合数，然后减去它，继续重复上述过程
    • 组合数计算：
      • 首先计算C（n,k）一定取最小的k，减少计算
      • 以C（8，5）为例子，分子只需计算8*7 *6 ，分母只需计算3 * 2 * 1
    #include <stdio.h>
#include <stdlib.h>
/**
 * 计算组合数 C(n, k) = n! / (k! * (n-k)!)
 * 使用迭代方法避免大数的直接计算和减少溢出的可能性。
 */
long long comb(int n, int k) {
    if (k > n) return 0;
    if (k * 2 > n) k = n - k; // 由于 C(n, k) = C(n, n-k)，取较小的 k 以减少计算量
    long long result = 1;
    for (int i = 1; i <= k; ++i, --n) {
        result = result * n / i; // 直接使用乘除法计算组合数，避免中间阶乘的大数
    }
    return result;
}

/**
 * 计算组合数分解的系数并填充到数组 c 中。
 * n 是要分解的数，k 是分解的项数，c 是存储系数的数组。
 */
void calcom(int n, int k, int c[]) {
    for (int i = k; i >= 1; i--) {
        int j = 1;
        while (comb(j, i) <= n) { // 查找满足条件的最大 j
            j++;
        }
        j--; // 回退到最后一个满足条件的 j
        c[k - i] = j; // 存储系数
        n -= comb(j, i); // 更新 n 为剩余的数
    }
}

int main() {
    int n, k;
    printf("请输入自然数n和k（格式为n,k）：");
    scanf("%d,%d", &n, &k); // 读取输入，期望格式为 n,k

    int c[k]; // 存储分解系数的数组
    calcom(n, k, c); // 计算组合数分解的系数
    
    // 输出分解结果
    printf("分解结果为：\n");
    for (int i = 0; i < k; i++) {
        printf("%d", c[i]);
        if (i < k - 1) printf(", "); // 在系数间添加逗号和空格
    }
    printf("\n");
    
    return 0;
}
